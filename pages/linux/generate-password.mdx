import { Steps, Tab, Tabs } from 'nextra-theme-docs'

# Generate a Password on Linux

<Tabs items={['OpenSSL', 'Password Gen']}>
  <Tab>
    <Steps>
    ### Install OpenSSL
    ```bash copy
    apt install openssl
    ```
    ### Generate Password
    ```bash copy
    openssl rand 60 | openssl base64 -A
    ```
    </Steps>
  </Tab>
  <Tab>
    <Steps>
    ### Install Pwgen
    ```bash copy
    apt install pwgen
    ```
    ### Generate Password
    ```bash copy
    pwgen -s 64 1
    ```
    </Steps>
  </Tab>
</Tabs>
